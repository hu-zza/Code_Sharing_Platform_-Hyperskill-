  let i = 0;
  
  function addTodoItem() {
    $("table")
      .append(`<tr id="task_${i}"></tr>`)
    
    $("table tr:last-child")
      .append(`<td onClick="toggleTaskMark(${i})">${$("input").val()}</td>`)
      .append(`<td onClick="deleteTask(${i})"><i class="fas fa-times-circle"></i></td>`)
      .append(`<td onClick="toggleTaskMark(${i})"><i class="fas fa-check-circle"></i></td>`);
    
    i++;
  }
  
  let toggleTaskMark = (id) => $(`#task_${id}`).toggleClass("done");
  let deleteTask = (id) => $(`#task_${id}`).remove();
